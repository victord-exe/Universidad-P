create table Actor(
id_actor number not null,
actor varchar(25) not null
CONSTRAINT actor_pk primary key (ID_actor)
);

create table Genero(
generoN varchar(25) not null,
id_genero number not null,

CONSTRAINT genero_pk primary key (ID_genero)
);

create table Pelicula (
p_nombre varchar(25) not null,
id_pelicula number not null,
id_genero_pel number not null,

CONSTRAINT pelicula_pk primary key (id_pelicula),

CONSTRAINT pelicula_fk foreign key (id_genero_pel)
references Genero(id_genero)
);

create table Pelicula_Actor(
id_actor_peliact number not null,
id_pelicula_peliact number not null,
personaje varchar(25) not null,
papel varchar(25) not null,

CONSTRAINT pelicula_actor_fk1 foreign key (id_actor_peliact)
References Actor(id_actor),

CONSTRAINT pelicula_Actor_fk2 foreign key (id_pelicula_peliact)
References Pelicula(id_pelicula),

CONSTRAINT pelicula_actor_pk primary key (id_actor_peliact, id_pelicula_peliact)
);
